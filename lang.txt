// ----------------
// CONTROL FLOW ?
// ----------------
// • `?` is the ONLY control-flow construct.
// • It pattern-matches on data
// • Exhaustive by default unless `else` is used
// • No if / switch / try / catch
// • cases begin with |

? hero.ai_state
| Idle: hero Rest()
| Patrol(dir): hero Move(dir)
    
// Optional unwrapping looks similar

? hero.target
| some(entity): hero Shoot(entity)
| none: hero Search()


// # introduces a temporary value to match on
// when you dont want to name it.

? # GetTotalAmmo(hero)
| # > 0: hero Shoot()
| else: hero Reload()


// Boolean is an enum with some syntax sugar.
// so you dont have to write out "| true:"

? _ CheckAmmo(hero)
    hero Shoot()
| else:
    hero Reload()

// ----------------
// INITIALIZATION
// ----------------
// Adding the keyword init will make it a contructor.
// it creates a value by mutation and the compiler will force:
// 1. All fields must be set
// 2. No field may be read before set
=== hero <- Init (data)
    hero.entity = data.entity
    hero.weapon = data.weapon
    hero.i_health = data.health

=== hero <- Init fromHealth(data, i_health)
    hero.entity = data.entity
    hero.weapon = data.weapon
    hero.i_health = i_health

=== Example
    new weapon
    use(weapon) // <- Compiler Error: not initialized

    //-----------------------------------------

    new weapon init(path)
    use(weapon)  // <- OK

    //-----------------------------------------

    new weapon

    ? hero.class
    | Warrior: 
        weapon Init(sword_data)
    | Archer:
        weapon Init(bow_data)

    hero Equip(weapon) // OK
